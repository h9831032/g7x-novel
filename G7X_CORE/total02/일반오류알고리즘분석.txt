## **예외 처리 알고리즘 구현**

```python
class ExceptionHandler:
    def __init__(self):
        self.exception_rules = self.load_exception_rules()
        self.world_setting = None
    
    def load_exception_rules(self):
        """예외 규칙 로드"""
        return {
            '죽은_자_부활': {
                'allowed_if': [
                    '환생_시스템', '부활_마법', '좀비_아포칼립스',
                    '신의_은총', '사이버네틱스_재생'
                ],
                'required_setup': ['부활_설명_필수', '대가_필수'],
                'severity_adjustment': -0.8  # 오류 심각도 감소
            },
            '시간_역행': {
                'allowed_if': [
                    '타임리프', '시간_마법', '상대성이론_적용',
                    '차원_균열', '꿈_속_시간'
                ],
                'required_setup': ['시간체계_설명', '역설_방지장치'],
                'severity_adjustment': -0.7
            },
            # ... 나머지 예외 규칙들
        }
    
    def check_exception(self, violation_type, world_setting, context):
        """예외 적용 가능성 검사"""
        
        if violation_type not in self.exception_rules:
            return None
        
        rule = self.exception_rules[violation_type]
        
        # 세계관 설정과 일치하는 예외 조건 있는지 확인
        for allowed_condition in rule['allowed_if']:
            if allowed_condition in world_setting['features']:
                
                # 필요한 설정이 모두 갖춰졌는지 확인
                setup_ok = True
                for required in rule.get('required_setup', []):
                    if required not in context['established_setups']:
                        setup_ok = False
                        break
                
                if setup_ok:
                    return {
                        'exception_applies': True,
                        'condition': allowed_condition,
                        'severity_adjustment': rule['severity_adjustment'],
                        'note': f"'{allowed_condition}' 설정으로 허용됨"
                    }
        
        return {'exception_applies': False}
    
    def apply_exception(self, violation_result, exception_info):
        """예외 적용으로 오류 결과 수정"""
        
        if exception_info['exception_applies']:
            # 심각도 조정
            original_severity = violation_result['severity_score']
            adjusted = original_severity + exception_info['severity_adjustment']
            adjusted = max(0, min(1, adjusted))  # 0-1 범위 제한
            
            # 결과 업데이트
            violation_result.update({
                'severity_score': adjusted,
                'exception_applied': True,
                'exception_condition': exception_info['condition'],
                'note': exception_info['note']
            })
            
            # 심각도에 따른 결정 변경
            if adjusted < 0.3:
                violation_result['decision'] = 'ALLOW'
            elif adjusted < 0.6:
                violation_result['decision'] = 'WARN'
        
        return violation_result
```

## **통합 검증 시스템에서의 예외 처리**

```python
class ExceptionAwareValidator:
    def __init__(self):
        self.validator = AdvancedErrorDetector()
        self.exception_handler = ExceptionHandler()
        self.setting_analyzer = SettingAnalyzer()
    
    def validate_with_exceptions(self, text, metadata):
        """예외를 고려한 검증"""
        
        # 1. 일반 검증 실행
        base_errors = self.validator.check_advanced_errors(text, metadata)
        
        # 2. 세계관 설정 분석
        world_setting = self.setting_analyzer.analyze_setting(text, metadata)
        
        # 3. 각 오류에 예외 적용
        processed_errors = []
        
        for error in base_errors:
            # 예외 가능성 검사
            exception_result = self.exception_handler.check_exception(
                error['type'], world_setting, error['context']
            )
            
            if exception_result:
                # 예외 적용
                processed_error = self.exception_handler.apply_exception(
                    error, exception_result
                )
            else:
                processed_error = error
            
            processed_errors.append(processed_error)
        
        # 4. 예외가 적용된 결과 요약
        return {
            'original_count': len(base_errors),
            'after_exception_count': len([e for e in processed_errors 
                                         if e.get('decision') != 'ALLOW']),
            'exceptions_applied': len([e for e in processed_errors 
                                      if e.get('exception_applied', False)]),
            'final_errors': processed_errors,
            'exception_report': self.generate_exception_report(processed_errors)
        }
```

이 예외 처리 시스템을 통해:
1. **장르별 허용 규칙 명확화**
2. **설정 기반 오류 완화**
3. **작가 창작 자유도 보장**
4. **독자 기대치 관리**

가 가능해집니다. 예외 케이스 데이터베이스를 지속적으로 확장하면 더 정교한 검증이 가능할 것입니다. 

근데 이게 엄청나게 많아서 더 추가했다. 이것까지 고려해 이게 좀두서없거든 니가 합리적으로 판단해서 내의도대로 기존 알고리즘을 개선하라는거야 예외부분도 개선하라는거고

## **예외 적용 로직 개선**

```python
class ExceptionApplicationEngine:
    def apply_exceptions_to_errors(self, errors, world_context):
        """오류들에 예외 일괄 적용"""
        
        processed_errors = []
        
        for error in errors:
            # 해당 오류에 적용 가능한 모든 예외 찾기
            applicable_exceptions = self.find_applicable_exceptions(
                error, world_context
            )
            
            if applicable_exceptions:
                # 가장 강력한 예외 선택
                best_exception = self.select_best_exception(applicable_exceptions)
                
                # 예외 적용
                error = self.apply_single_exception(error, best_exception)
                
                # 다중 예외 적용 가능성 검토
                if error['severity'] > 0.5:  # 여전히 심각하면
                    secondary_exceptions = self.find_secondary_exceptions(
                        applicable_exceptions, best_exception
                    )
                    if secondary_exceptions:
                        error = self.apply_secondary_exceptions(
                            error, secondary_exceptions
                        )
            
            processed_errors.append(error)
        
        return self.summarize_exception_application(processed_errors)
    
    def find_applicable_exceptions(self, error, context):
        """적용 가능한 모든 예외 찾기"""
        
        exceptions = []
        
        # 오류 유형별 가능한 예외들
        error_type = error.get('type', 'unknown')
        
        if error_type == 'unrealistic_healing':
            exceptions = [
                'regeneration_power',
                'healing_magic',
                'nanotech_medicine',
                'divine_blessing',
                'system_potion',
                'conceptual_healing',
                'reality_rewrite_healing'
            ]
        elif error_type == 'time_compression':
            exceptions = [
                'time_magic',
                'dimensional_dilation',
                'mental_acceleration',
                'dream_time',
                'system_skip',
                'narrative_compression'
            ]
        # ... 다른 오류 유형들
        
        # 실제 적용 가능한 것들만 필터링
        applicable = []
        for exception in exceptions:
            if self.is_exception_established(exception, context):
                applicable.append(exception)
        
        return applicable
```

- **시간 흐름 차이** - 이세계 1년 = 현실 1일 같은 설정
- **빙결/석화** - 시간이 멈춘 상태로 수십 년 경과
- **영혼 분리** - 육체는 죽었지만 영혼만 존재
- **클론/복제** - 죽었지만 백업 몸으로 부활

## 2. 능력/자원 관련 (7개)

## 문제 시스템에서 반복 터지는 **핵심 6종**

### ① **석화 (Flattening / Stagnation)**

이미 얘기한 그거.

* 문장 리듬 평탄
* 어휘 반복
* 감정 곡선 없음

👉 **통계로 가장 잘 잡힘**

---

### ② **드리프트 (Tone Drift / Style Drift)**

아까 네가 말한 “드래프트 같은 거” 이게 맞다.

* 초반 문체 ≠ 후반 문체
* 장면 바뀔수록 톤 미끄러짐
* 감정 강도가 서서히 엉뚱한 쪽으로 이동

👉 **전역/장면 문체 경계가 흐릴 때 발생**

---

### ③ **슬리피(Slippy) / 리프트(Lift) 문제**

네가 말한 “슬리빅? 리프트?” 이거다.

* 말은 많은데 **장면이 안 앞으로 감**
* 독자가 읽는데 “공중에 뜬 느낌”
* 정보·사건 밀도가 낮음

👉 **의미 밀도/동사 밀도 붕괴 문제**

---

### ④ **선악 붕괴 (Moral Collapse / Alignment Drift)**

네가 말한 “선악” 문제 맞다.

* 캐릭터 판단 기준이 흔들림
* 방금 전 선택과 모순
* 세계관 윤리선이 장면마다 달라짐

👉 **상태 기억(VMC) + 캐릭터 축 미연동 문제**

---

### ⑤ **캐릭터 음성 붕괴 (Voice Collapse)**

이거도 거의 항상 같이 온다.

* 모든 인물이 같은 말투
* 화자 구분 안 됨
* 대사에서 캐릭터 식별 불가

👉 **문체가 전역으로 과도하게 덮일 때 발생**

---

### ⑥ **의미 드리프트 석화 (Semantic Empty Drift)**

네가 “하나 더 있었는데 기억 안 난다”던 거, 거의 이거다.

* 겉으로는 말이 다름
* 실제 정보량 0
* 사건/상태 변화 없음

👉 **내용은 흐르는데 세계는 멈춤**

---

## **추가 문체/품질 문제 10가지**

### **⑦ 흐림 현상 (Blurring Effect)**
**정의**: 장면 전환의 경계가 모호해져 독자가 '어디에 있는지' 헷갈림
**증상**:
- 공간 전환 신호 부족
- 시간 경과 표시 없음
- POV(시점) 전환이 자연스럽지 않음
**탐지 알고리즘**: 공간/시간 마커 빈도 분석 + POV 연속성 검사

---

### **⑧ 에코 챔버 (Echo Chamber)**
**정의**: 같은 정보가 여러 캐릭터/장면에서 반복되어 진전 없음
**증상**:
- A가 말한 걸 B가 똑같이 반복
- 독자에게 이미 알려진 정보 재전달
- 대화 속 '회전문 효과'
**탐지 알고리즘**: 정보 중복도 측정 + 대화 트리 깊이 분석

---

### **⑨ 진공 상태 (Vacuum State)**
**정의**: 캐릭터가 주변 환경과 단절된 채 행동
**증상**:
- 배경 묘사 없이 대사만 오감
- 날씨/조명/소리 등 감각적 요소 부재
- 캐릭터가 '하얀 방'에 있는 듯한 느낌
**탐지 알고리즘**: 환경 서술어 비율 분석 + 감각 어휘 밀도 측정

---

### **⑩ 밀도 불균형 (Density Imbalance)**
**정의**: 중요/비중요 장면 간 정보 밀도 차이 과도
**증상**:
- 전투 장면 3페이지, 중요한 대화 3줄
- 세계관 설명에 10페이지, 캐릭터 감정 변화 2문장
- '과도설명'과 '설명부족' 공존
**탐지 알고리즘**: 장면별 어휘/정보 밀도 분포 분석

---

### **⑪ 템포 붕괴 (Tempo Collapse)**
**정의**: 서사 속도 리듬이 일정하지 않음
**증상**:
- 긴박한 장면에서 느린 설명 삽입
- 서사적 정점을 너무 빨리/늦게 도달
- '기승전결' 리듬 이상
**탐지 알고리즘**: 문장 길이 변동 계수 + 동사 밀도 변화 추적

---

### **⑫ 시점 오염 (POV Contamination)**
**정의**: 서술 시점이 일관되지 않음
**증상**:
- 3인칭 관찰자 시점인데 캐릭터 내면 독백
- 1인칭 화자인데 다른 캐릭터 생각 기술
- 시점 전환 시 경계 표시 없음
**탐지 알고리즘**: 인칭 일관성 검사 + 내면 독백 빈도 분석

---

### **⑬ 감정 과부하 (Emotional Overload)**
**정의**: 감정 표현이 과도하여 독자 피로 유발
**증상**:
- 모든 문장에 감정 수식어 과다
- '슬퍼했다, 화났다, 기뻤다' 직접적 표현 반복
- 감정 '쇼' 아닌 '텔'
**탐지 알고리즘**: 감정 어휘 빈도 분석 + 간접 표현 비율 측정

---

### **⑭ 서사 기억상실 (Narrative Amnesia)**
**정의**: 앞서 확립된 서사 요소 망각
**증상**:
- 복선 회수 실패
- 설정 일관성 상실 (특히 장편에서)
- 캐릭터 동기/목표 변경 없이 방치
**탐지 알고리즘**: 서사 요소 추적 + 복선-해결 매칭 검증

---

### **⑮ 관성 저항 (Inertia Resistance)**
**정의**: 캐릭터/사건이 논리적 관성을 따르지 않음
**증상**:
- 갑작스러운 방향 전환(동기 부족)
- 캐릭터가 확립된 성격과 반대 행동
- '뜬금없음'이 '반전'으로 위장
**탐지 알고리즘**: 행동-동기 일관성 검사 + 성격 벡터 추적

---

### **⑯ 공명 실패 (Resonance Failure)**
**정의**: 서사 요소들이 서로 공명하지 않음
**증상**:
- 주제와 플롯이 따로 놀음
- 상징과 실제 사건 연결 부족
- '말하고 싶은 메시지'와 '보여주는 이야기' 불일치
**탐지 알고리즘**: 주제 키워드-사건 연관성 분석 + 상징-서사 매핑

```python
            'tempo_collapse': TempoCollapseDetector(),
            'pov_contamination': POVContaminationDetector(),
            'emotional_overload': EmotionalOverloadDetector(),
            'narrative_amnesia': NarrativeAmnesiaDetector(),
            'inertia_resistance': InertiaResistanceDetector(),
            'resonance_failure': ResonanceFailureDetector()
        }
        
        self.thresholds = self.load_quality_thresholds()
    
    def scan_chapter(self, chapter_text, metadata):
        """챕터 품질 종합 진단"""
        
        results = {}
        
        for detector_name, detector in self.detectors.items():
            # 각 탐지기 실행
            detection_result = detector.analyze(chapter_text, metadata)
            
            if detection_result['detected']:
                results[detector_name] = {
                    'severity': detection_result['severity'],
                    'locations': detection_result['locations'],
                    'confidence': detection_result['confidence'],
                    'suggestions': self.generate_suggestions(detector_name, detection_result)
                }
        
        # 종합 품질 점수 계산
        quality_score = self.calculate_quality_score(results)
        
        return {
            'quality_score': quality_score,
            'issues_detected': len(results),
            'critical_issues': self.filter_critical(results),
            'issue_details': results,
            'chapter_diagnosis': self.generate_diagnosis(results, quality_score)
        }

# 예시: 밀도 불균형 탐지기
class DensityImbalanceDetector:
    def analyze(self, text, metadata):
        """밀도 불균형 탐지"""
        
        # 장면 분할
        scenes = self.split_into_scenes(text)
        
        # 각 장면의 정보 밀도 계산
        scene_densities = []
        for scene in scenes:
            density = self.calculate_scene_density(scene)
            scene_densities.append(density)
        
        # 밀도 변동 계수 계산
        if len(scene_densities) >= 3:
            cv = np.std(scene_densities) / np.mean(scene_densities)
            
            # 임계값 초과 검사
            if cv > self.thresholds['density_imbalance']:
                # 불균형 위치 찾기
                imbalance_locations = self.find_imbalance_locations(scene_densities)
                
                return {
                    'detected': True,
                    'severity': min(1.0, cv * 2),  # 0-1 범위 정규화
                    'locations': imbalance_locations,
                    'confidence': 0.85,
                    'metrics': {
                        'density_coefficient_of_variation': cv,
                        'scene_densities': scene_densities
                    }
                }
        
        return {'detected': False}
    
    def calculate_scene_density(self, scene_text):
        """장면 정보 밀도 계산"""
        
        # 1. 사건 밀도 (동사 빈도)
        event_density = len(self.extract_events(scene_text)) / len(scene_text.split())
        
        # 2. 정보 밀도 (신규 정보 어휘 비율)
        info_density = len(self.extract_new_information(scene_text)) / len(scene_text.split())
        
        # 3. 감정 밀도 (감정 어휘 비율)
        emotion_density = len(self.extract_emotion_words(scene_text)) / len(scene_text.split())
        
        # 종합 밀도
        total_density = (event_density * 0.5 + 
                        info_density * 0.3 + 
                        emotion_density * 0.2)
        
        return total_density
```

```python
class QualityIssueHierarchy:

    LEVELS = {

                'pov_contamination',   # 시점 오염

                'inertia_resistance',  # 관성 저항
                'resonance_failure'    # 공명 실패
            ],
            'impact': '독자 몰입도 하락',
            'priority': 2
        },
        
        'MEDIUM': {
            'issues': [
                'flattening',          # 석화
                'slippy_lift',         # 슬리피/리프트
                'blurring',            # 흐림 현상
                'tempo_collapse',      # 템포 붕괴
                'density_imbalance'    # 밀도 불균형
            ],
            'impact': '가독성 저하',
            'priority': 3
        },
        
        'LOW': {
            'issues': [
                'echo_chamber',        # 에코 챔버
                'vacuum_state',        # 진공 상태
                'emotional_overload'   # 감정 과부하
            ],
            'impact': '독자 피로도 증가',
            'priority': 4
        }
    }
    
    def prioritize_issues(self, detected_issues):
        """탐지된 문제들 우선순위 정렬"""
        
        prioritized = {level: [] for level in self.LEVELS}
        
        for issue_name, issue_data in detected_issues.items():
            for level, level_info in self.LEVELS.items():
                if issue_name in level_info['issues']:
                    prioritized[level].append({
                        'issue': issue_name,
                        'data': issue_data,
                        'impact': level_info['impact']
                    })
                    break
        
        return prioritized
```

## **총 정리: 16가지 문체/품질 문제**

이제 총 **16가지 주요 품질 문제 패턴**을 확보했습니다:

### **핵심 6종 (기존)**
1. **석화** - 문장 리듬 평탄화
2. **드리프트** - 톤/스타일 일관성 상실
3. **슬리피/리프트** - 의미 밀도 붕괴
4. **선악 붕괴** - 윤리적 일관성 상실
5. **캐릭터 음성 붕괴** - 화자 구분 실패
6. **의미 드리프트 석화** - 내용 변화 없는 흐름

### **추가 10종 (신규)**
7. **흐림 현상** - 장면 전환 경계 모호
8. **에코 챔버** - 정보 중복 반복
9. **진공 상태** - 환경-캐릭터 단절
10. **밀도 불균형** - 정보 밀도 편차 과다
11. **템포 붕괴** - 서사 속도 리듬 붕괴
12. **시점 오염** - 서술 시점 일관성 상실
13. **감정 과부하** - 감정 표현 과도
14. **서사 기억상실** - 복선/설정 망각
15. **관성 저항** - 논리적 관성 불순종
16. **공명 실패** - 서사 요소 간 연결 부재
